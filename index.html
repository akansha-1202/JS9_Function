QUESTION 2. 

Create one function which takes two values as a parameter and print the sum of them as "Sum of 3, 4 is 7"

SOLUTION. 
function addTwoNo(a,b){
        let sum=a+b;
        console.log("Sum of Two No:"+sum);
      }
      addTwoNo(3,4);
Input: a=3 b=4
Output: Sum of Two No:7



QUESTION 3. 

Create one arrow function

SOLUTION. 
const addTwoNo=(A,B)=>{
        let sum=A+B;
        console.log("Sum of Two No:"+sum);
      }
      addTwoNo(3,4)
Input: A=3 B=4
Output: Sum of Two No:7

QUESTION 7. 

Write a function that accepts parameter "N" and returns factorial of "N".

SOLUTION. 
        function findFactorial(N){
            let result=1;
            for(let i=1;i<=N;i++){
              result*=i;
            }
            return result;
        };

        let N = 5;
        let factorial = findFactorial(N);
        console.log(factorial);
Output: 120
